country_event = {
	# POPULAR FRONT VICTORY IN 1936 ELECTION
	# Player chooses here whether they will side with the Nationalists or the Republicans in the civil war
	id = spain_civil_war_prelude.1
	title = spain_civil_war_prelude.1.title
	desc = spain_civil_war_prelude.1.desc

	is_triggered_only = yes
	fire_only_once = yes
	timeout_days = 7

	option = {
		name = spain_civil_war_prelude.1.a # Side with the Republic

		custom_effect_tooltip = spain_civil_war_play_as_republicans_tt
		set_country_flag = spain_civil_war_play_as_republicans_flag

		add_stability = -0.1
	}

	option = {
		name = spain_civil_war_prelude.1.b # Side with the Nationalists

		custom_effect_tooltip = spain_civil_war_play_as_nationalists_tt
		set_country_flag = spain_civil_war_play_as_nationalists_flag

		add_stability = -0.1
	}

	after = {
		country_event = {
			id =  spain_civil_war_prelude.2
			days = 10
		}
	}
}

country_event = {
	# RIGHT WITHDRAWS FROM THE PROCESS IN PROTEST
	id = spain_civil_war_prelude.2
	title = spain_civil_war_prelude.2.title
	desc = spain_civil_war_prelude.2.desc

	is_triggered_only = yes
	fire_only_once = yes
	timeout_days = 7

	## Options if the player is siding with the Republic
	option = {
		# Defend Republicanism
		trigger = {
			has_country_flag = spain_civil_war_play_as_republicans_flag
		}

		name = spain_civil_war_prelude.2.a

		VX_increase_republican_strength_effect = yes
		add_popularity = {
			ideology = progressive_ideology
			popularity = 1
		}
		add_popularity = {
			ideology = social_democracy_ideology
			popularity = 2
		}
		add_popularity = {
			ideology = liberal_ideology
			popularity = 2
		}
	}

	option = {
		# Support Radicals
		trigger = {
			has_country_flag = spain_civil_war_play_as_republicans_flag
		}

		name = spain_civil_war_prelude.2.b
		
		VX_increase_socialist_strength_effect = yes
		add_popularity = {
			ideology = revolutionary_socialism_ideology
			popularity = 3
		}
		add_popularity = {
			ideology = authoritarian_socialism_ideology
			popularity = 2
		}
	}

	option = {
		# Support Neither
		trigger = {
			has_country_flag = spain_civil_war_play_as_republicans_flag
		}

		name = spain_civil_war_prelude.2.c		

		VX_increase_overall_strength_effect = yes
	}

	## Options if the player sided with the Nationalists

	option = {
		# Strengthen the Military
		trigger = {
			has_country_flag = spain_civil_war_play_as_nationalists_flag
		}

		name = spain_civil_war_prelude.2.44DescIsPuppet

		VX_increase_military_strength_effect = yes
		add_popularity = {
			ideology = autocracy_ideology
			popularity = 5
		}
	}

	option = {
		# Strengthen Falangists
		trigger = {
			has_country_flag = spain_civil_war_play_as_nationalists_flag
		}

		name = spain_civil_war_prelude.2.e

		VX_increase_falangist_strength_effect = yes
		add_popularity = {
			ideology = fascist_ideology
			popularity = 5
		}
	}

	option = {
		
		trigger = {
			has_country_flag = spain_civil_war_play_as_nationalists_flag
		}	

		name = spain_civil_war_prelude.2.f

		VX_increase_overall_strength_effect = yes
	}

	after = {
		country_event = {
			id =  spain_civil_war_prelude.3
			days = 8
		}
	}
}
