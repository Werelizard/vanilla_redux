on_actions = {
	on_startup = {
		# Big startup chunk
		effect = {
			#
			log = "=== VANILLA REDUX ==="
			every_possible_country = {				
				# Add opinion modifiers pertaining to government ideologies
				limit = { 
					NOT = { 
						tag = event_target:system
					}
				}
				startup_opinion_modifiers_effect = yes				
				# Initiate coalition mechanics for every country
				add_dynamic_modifier = {
					modifier = coalitions_dynmod
				}
				update_coalitions_effect = yes
				# Create arrays for their cores at the start smile emoji
				store_core_states_on_game_start_effect = yes
			}
		}
	}
	on_government_change = {
		# Update the opinion modifiers pertaining to government ideologies
		effect = {
			update_opinion_modifiers_effect = yes
		}
		# Update the coalitions
		effect = {
			update_coalitions_effect = yes
			remove_uncessary_coalitions_effect = yes # Called here to prevent redundant coalition flags
		}
	}
	on_daily = {
		# This MUST be daily for coalitions to work
		effect = {			
			if = {
				limit = {
					any_coalition_active_trigger = yes
				}
				update_coalitions_effect = yes
			}
		}
	}
	on_new_term_election = {		
		# countries with content should have their election events in their own "on_action" files
		effect = {
			if = {
				limit = {
					NOT = {
						has_VX_content_trigger = yes
					}
				}
			}
		}
	}
	on_peaceconference_started = {
		# Multi-Purpose Flags to determine if the country has defeated a major with a given ideology
		effect = {
			if = {
				limit = {
					FROM = {
						has_fascist_government_trigger = yes
						is_major = yes
					}
					ROOT = {
						NOT = {
							has_fascist_government_trigger = yes
						}
						NOT = {
							has_country_flag = defeated_fascist_major_flag
						}
					}
				}
				ROOT = {
					set_country_flag = defeated_fascist_major_flag
				}
			}
			if = {
				limit = {
					FROM = {
						has_socialist_government_trigger = yes
						is_major = yes
					}
					ROOT = {
						NOT = {
							has_socialist_government_trigger = yes
						}
						NOT = {
							has_country_flag = defeated_socialist_major_flag
						}
					}
				}
				ROOT = {
					set_country_flag = defeated_socialist_major_flag
				}
			}
			if = {
				limit = {
					FROM = {
						has_democratic_government_trigger = yes
						is_major = yes
					}
					ROOT = {
						NOT = {
							has_democratic_government_trigger = yes
						}
						NOT = {
							has_country_flag = defeated_democratic_major_flag
						}
					}
				}
				ROOT = {
					set_country_flag = defeated_democratic_major_flag
				}
			}
			if = {
				limit = {
					FROM = {
						has_authoritarian_government_trigger = yes
						is_major = yes
					}
					ROOT = {
						NOT = {
							has_authoritarian_government_trigger = yes
						}
						NOT = {
							has_country_flag = defeated_authoritarian_major_flag
						}
					}
				}
				ROOT = {
					set_country_flag = defeated_authoritarian_major_flag
				}
			}
		}
	}
}
