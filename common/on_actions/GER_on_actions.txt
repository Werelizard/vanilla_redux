		effect = {
			GER = {
				# Removes the dummy ideas for germany
				remove_ideas = ger_dummy_MEFO

				remove_ideas = ger_dummy_army
				remove_ideas = ger_dummy_navy
				remove_ideas = ger_dummy_airforce
			}
		}
	}
	# Effect to update the "Leader of the New Order" dynmod
	on_join_faction = {
		effect = {
			FROM = {
				if = {
					limit = {
						original_tag = GER
						has_dynamic_modifier = GER_new_order_dynmod
					}
				}
			}
		}
	}
}
