on_actions = {
	on_startup = {
		effect = {
			every_possible_country = { # Make sure the coalitions actually work
				add_dynamic_modifier = { 
					modifier = coalitions_glob_dynmod
				}
				update_coalitions_effect = yes
			}
			every_possible_country = {
				# Effect set up the opinion modifiers
				limit = { 
					NOT = { 
						tag = event_target:system
					}
				}
				VX_startup_ideology_impact_effect = yes
			}
		}
	}
	on_daily = {
		effect = {
			every_possible_country = { # make sure to upload the coalitions every day
				update_coalitions_effect = yes
			}
		}
	}
	on_government_change = {
		effect = {
			VX_update_ideology_impact_effect = yes
		}
	}
}
