# Update the coalitions
update_coalitions_effect = {
	set_variable = {
		coalition_effect_var = 0
	}
	if = {
		limit = {
			authsoc_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@authoritarian_socialism_ideology
		}
	}
	if = {
		limit = {
			revsoc_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@revolutionary_socialism_ideology
		}
	}
	if = {
		limit = {
			socdem_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@social_democracy_ideology
		}
	}
	if = {
		limit = {
			progressive_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@progressive_ideology
		}
	}
	if = {
		limit = {
			liberal_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@liberal_ideology
		}
	}
	if = {
		limit = {
			conservative_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@conservatism_ideology
		}
	}
	if = {
		limit = {
			authoritarian_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@authoritarian_democrat_ideology
		}
	}
	if = {
		limit = {
			autocracy_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@autocracy_ideology
		}
	}
	if = {
		limit = {
			fascism_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@fascist_ideology
		}
	}
	add_to_variable = {
		coalition_effect_popularity_var = coalition_effect_var
	}
	add_to_variable = {
		coalition_effect_political_var = coalition_effect_var
	}
	divide_variable = {
		coalition_effect_political_var = 5
	}
	divide_variable = {
		coalition_effect_popularity_var = 2.5
	}
}
