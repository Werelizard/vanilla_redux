# Opinion related effects
startup_opinion_modifiers_effect = {
	every_possible_country = {
		limit = {
			NOT = {
				tag = PREV
			}
		}
		# If they have the same ideology within the same group
		if = {
			limit = {
				has_government = PREV
			}
			add_opinion_modifier = {
				target = PREV
				modifier = VX_same_ideology
			}
		}
		else_if = {
			# If both parties are within the same group, but do not have the same ideology
			limit = {
				OR = {
					AND = {
						# If both countries are democratic
						PREV = {
							has_socialist_government_trigger = yes
						}
						has_socialist_government_trigger = yes
					}
					AND = {
						# If both countries are socialist
						PREV = {
							has_democratic_government_trigger = yes
						}
						has_democratic_government_trigger = yes
					}
					AND = {
						# If both countries are authoritarian
						PREV = {
							has_authoritarian_government_trigger = yes
						}
						has_authoritarian_government_trigger = yes
					}
					AND = {
						# If both countries are fascist
						PREV = {
							has_fascist_government_trigger = yes
						}
						has_fascist_government_trigger = yes
					}
				}
			}
			add_opinion_modifier = {
				target = PREV
				modifier = VX_same_ideology_group
			}
		}
		else_if = {
			# For "similar" cases
			limit = {
				OR = {
					AND = {
						# If they're both conservative or authoritarian
						PREV = {
							OR = {
								has_government = conservative
								has_authoritarian_government_trigger= yes
							}
						}
						OR = {
							has_government = conservative
							has_authoritarian_government_trigger = yes
						}
					}
					AND = {
						# If they're both revolutionary socialist or social-democratic
						PREV = {
							OR = {
								has_government = revolutionary_socialist
								has_government = social_democratic
							}
						}
						OR = {
							has_government = revolutionary_socialist
							has_government = social_democratic
						}
					}
				}
			}
			add_opinion_modifier = {
				target = PREV
				modifier = VX_similar_ideology
			}
		}
		else = {
			add_opinion_modifier = {
				target = PREV
				modifier = VX_different_ideology
			}
		}
	}
}

# Update the coalitions
update_coalitions_effect = {
	set_variable = {
		coalition_effect_var = 0
	}
	if = {
		limit = {
			authsoc_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@authoritarian_socialist
		}
	}
	if = {
		limit = {
			revsoc_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@revolutionary_socialist
		}
	}
	if = {
		limit = {
			socdem_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@social_democratic
		}
	}
	if = {
		limit = {
			progressive_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@progressive
		}
	}
	if = {
		limit = {
			liberal_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@liberal
		}
	}
	if = {
		limit = {
			conservative_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@conservative
		}
	}
	if = {
		limit = {
			authoritarian_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@authoritarian_democrat
		}
	}
	if = {
		limit = {
			autocracy_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@autocratic
		}
	}
	if = {
		limit = {
			fascist_coalition_trigger = yes
		}
		add_to_variable = {
			coalition_effect_var = party_popularity@fascist
		}
	}
	add_to_variable = {
		coalition_small_effect_var = coalition_effect_var
	}
	add_to_variable = {
		coalition_big_effect_var = coalition_effect_var
	}
	divide_variable = {
		coalition_small_effect_var = 5
	}
	divide_variable = {
		coalition_big_effect_var = 2.5
	}
}
