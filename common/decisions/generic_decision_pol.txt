generic_decision_war_measures = {
	# War propaganda
	generic_war_propaganda = {
		visible = {
			OR = {
				has_war_with_non_exiled_country = yes
				any_neighbor_country = {
					has_war = yes
				}
				AND = {
					OR = {
						has_fascist_government_trigger = yes
						has_socialist_government_trigger = yes
					}
					threat > 0.1
				}
				AND = {
					OR = {
						has_democratic_government_trigger = yes

				}
			}

		###
		available = {

		###
		modifier = {
			war_support_weekly = 0.005
		}
		###
		cancel_trigger = {
			has_war_support > 0.5
		}
		###
		ai_hint_pp_cost = 100
		ai_will_do = {
			factor = 0
			# AI should not take this decision if war support at 40% or higher            
			modifier = {
				add = -200
				has_war_support > 0.39
			}
			# AI should not take this decision if it can (and should) move to war economy
			modifier = {
				add = -200
				#                
				OR = {
					AND = {
						NOT = {
							has_idea = low_economic_mobilisation
						}
						any_enemy_country = {
							ic_ratio = {
								tag = ROOT
								ratio > 0
							}
						}
					}
					AND = {
						NOT = {
							has_idea = partial_economic_mobilisation
						}
						any_enemy_country = {
							ic_ratio = {
								tag = ROOT
								ratio > 0.2
							}
						}
					}
					AND = {
						NOT = {
							has_idea = war_economy
						}
						NOT = {
							has_idea = partial_economic_mobilisation
						}
						any_enemy_country = {
							ic_ratio = {
								tag = ROOT
								ratio > 0.5
							}
						}
					}
				}
			}
			# AI should not take this decision if it can (and should) increase its recruitement law
			modifier = {
				add = -200
				#                
				OR = {
					AND = {
						any_other_country = {
							has_war_with = ROOT
							fighting_army_strength_ratio = {
								tag = ROOT
								ratio > 0
							}
						}
						OR = {
							has_idea = disarmed_nation
							has_idea = volunteer_only
						}
					}
					AND = {
						any_other_country = {
							has_war_with = ROOT
							fighting_army_strength_ratio = {
								tag = ROOT
								ratio > 0.2
							}
						}
						OR = {
							has_idea = limited_conscription
						}
					}
					AND = {
						any_other_country = {
							has_war_with = ROOT
							fighting_army_strength_ratio = {
								tag = ROOT
								ratio > 0.5
							}
						}
						OR = {
							has_idea = service_by_requirement
						}
					}
				}
			}
			# AI should not take this if it doesn't have its high command and chiefs filled - unless its war support is really, really bad
			modifier = {
				add = -200
				OR = {
					has_war_support > 0.19
						# amount_taken_ideas = {
						# 	amount < 1
						# 	slots = {
						# 		army_chief
						# 	}
						# }
						# amount_taken_ideas = {
						# 	amount < 1
						# 	slots = {
						# 		air_chief
						# 	}
						# }
						# amount_taken_ideas = {
						# 	amount < 1
						# 	slots = {
						# 		navy_chief
						# 	}
						# }
						# amount_taken_ideas = {
						# 	amount < 1
						# 	slots = {
						# 		high_command
						# 	}
						# }
				}
			}
			# Final weights to add in depending on how much war support the AI has
			modifier = {
				add = 1
				has_war_support < 0.4
			}
			modifier = {
				add = 2
				has_war_support < 0.3
			}
			modifier = {
				add = 4
				has_war_support < 0.2
			}
		}
	}
