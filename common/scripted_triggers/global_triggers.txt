		original_tag = GER

	}
}

# Government related triggers

has_socialist_goverment_trigger = {
	OR = {
		has_government = authoritarian_socialist
		has_government = revolutionary_socialist
	}
}

has_democratic_goverment_trigger = {
	OR = {
		has_government = social_democratic
		has_government = progressive
		has_government = liberal
		has_government = liberal
	}
}

has_authoritarian_goverment_trigger = {
	OR = {
		has_government = authoritarian_democrat
		has_government = autocratic
	}
}

has_fascist_goverment_trigger = {
	OR = {
		has_government = fascist
	}
}

# Coalition related triggers

	custom_trigger_tooltip = {
		tooltip = authsoc_coalition_tt

			has_government = authoritarian_socialist
		}
