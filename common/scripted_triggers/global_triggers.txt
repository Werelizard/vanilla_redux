# Ideology relations triggers

is_enemy_ideology = {
	OR = {
		AND = {
			# The target country is SOCIALIST
			has_socialist_government_trigger = yes
			# Fascist countries, authoriraian anc conservative countries would consider it an enemy
			ROOT = {
				OR = {
					has_government = conservative
					has_authoritarian_government_trigger = yes
					has_fascist_government_trigger = yes
				}
			}
		}
		AND = {
			# The target country is DEMOCRATIC 
			has_democratic_government_trigger = yes
			# Communist and Fascist countries would consider democracies hostile
			OR = {
				AND = {
					# Socialist countries are only hostile if the democracy is NOT social-democratic
					NOT = {
						has_government = social_democratic
					}
					ROOT = {
						has_socialist_government_trigger = yes
					}
				}
				ROOT = {
					has_fascist_government_trigger = yes
				}
			}
		}
		AND = {
			# The target country is AUTHORITARIAN
			has_authoritarian_government_trigger = yes
			# Socialists and social democrats are hostile to authoritarian countries
			ROOT = {
				OR = {
					has_socialist_government_trigger = yes
					has_government = social_democratic
				}
			}
		}
		AND = {
			# The target country is fascist
			has_fascist_government_trigger = yes
			# Socialists, democracies are hostile to fascism
			ROOT = {
				OR = {
					has_socialist_government_trigger = yes
					has_democratic_government_trigger = yes
				}
			}
		}
	}
}

is_friendly_ideology = {
	OR = {
		AND = {
			# Target is SOCIALIST
			OR = {
				has_socialist_government_trigger = yes
				has_government = social_democratic
			}
			# Social democrats and other socialist countries should be friendly towards socialist countries and vice versa
			ROOT = {
				OR = {
					has_socialist_government_trigger = yes
					has_government = social_democratic
				}
			}
		}
		AND = {
			# Target is demoratic
			has_democratic_government_trigger = yes
			# Auth. Dem and democracies should be friendly towards democracies
			ROOT = {
				OR = {
					has_democratic_government_trigger = yes
					has_government = authoritarian_democrat
				}
			}
		}
		AND = {
			# Target is authoritarian but NOT autocracy
			has_authoritarian_government_trigger = yes
			NOT = {
				has_government = autocracy
			}
			# Conservatives and other authoritarians should be friendly
			ROOT = {
				OR = {
					has_authoritarian_government_trigger = yes
					has_government = conservative
				}
			}
		}
		AND = {
			# Target is an autocratic government
			has_government = autocracy
			# Fascist countires and authoritarian democrats are fond of military/totalist countries
			ROOT = {
				OR = {
					has_authoritarian_government_trigger = yes
					has_fascist_government_trigger = yes
				}
			}
		}
		AND = {
			# Target is a fascist country
			has_fascist_government_trigger = yes
			# Autocracies and fascist countries align with fascist nations
			ROOT = {
				OR = {
					has_government = autocracy
					has_fascist_government_trigger = yes
				}
			}
		}
	}
}

# This specifically overrides the rule check as to whether or not you can guarantee other ideologies
DIPLOMACY_GUARANTEE_ENABLE_TRIGGER_OVERRIDES_GAME_TRIGGER = {
	NOT = {
		has_game_rule = {
			rule = allow_guarantees
			option = LIMITED
		}
	}
}

DIPLOMACY_GUARANTEE_ENABLE_TRIGGER = {
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = ALWAYS_FREE
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = LIMITED
			}
		}
		always = yes # Handled by game logic.
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = SAME_IDEOLOGY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_GUARANTEES_SAME_IDEOLOGY_TOOLTIP
			has_same_government = yes
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = BLOCKED
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_GUARANTEES_BLOCKED_TOOLTIP
			always = no
		}
	}
}
