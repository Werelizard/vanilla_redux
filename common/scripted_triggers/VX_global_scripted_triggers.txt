### IDEOLOGY GROUP TRIGGERS ###
has_socialist_government_trigger = {
	custom_trigger_tooltip = {
		tooltip = has_ideology_socialist_tt
		OR = {
			has_government = authoritarian_socialism_ideology
			has_government = revolutionary_socialism_ideology
		}
	}
}

has_democratic_government_trigger = {
	custom_trigger_tooltip = {
		tooltip = has_ideology_democratic_tt
		OR = {
			has_government = social_democracy_ideology
			has_government = progressive_ideology
			has_government = liberal_ideology
			has_government = conservatism_ideology
		}
	}
}

has_authoritarian_government_trigger = {
	custom_trigger_tooltip = {
		tooltip = has_ideology_authoritarian_tt
		OR = {
			has_government = authoritarian_democrat_ideology
			has_government = autocracy_ideology
		}
	}
}

has_fascist_government_trigger = {
	custom_trigger_tooltip = {
		tooltip = has_ideology_nationalist_tt
		has_government = fascist_ideology
	}
}

# MAJORITY TRIGGER
VX_has_plurality_trigger = {
	if = {
		limit = {
			has_government = social_democracy_ideology
		}
		social_democracy_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = progressive_ideology
		}
		progressive_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = liberal_ideology
		}
		liberal_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = conservatism_ideology
		}
		conservatism_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = authoritarian_democrat_ideology
		}
		authoritarian_democrat_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = autocracy_ideology
		}
		autocracy_ideology > 0.2
	}
	else_if = {
		limit = {
			has_government = fascist_ideology
		}
		fascist_ideology > 0.2
	}
	else = {
		always = no
	}
}
